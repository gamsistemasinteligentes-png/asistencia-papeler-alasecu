function registrar(tipo) { const emp = document.getElementById('employee').value; if(!emp) return alert("Selecciona tu nombre"); const ahora = new Date(); const hora = ahora.toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}); const fecha = ahora.toLocaleDateString('es-MX', {day:'2-digit', month:'short'}); const ts = ahora.getTime(); const obs = document.getElementById('txtObs').value || "Sin notas"; const reg = { emp, tipo, hora, fecha, obs, ts }; // GUARDAR LOCALMENTE let logs = JSON.parse(localStorage.getItem('secu_vFinal')) || []; logs.unshift(reg); localStorage.setItem('secu_vFinal', JSON.stringify(logs.slice(0,30))); renderHistory(); // GENERAR TEXTO const texto = `🏢 *LA SECU - ASISTENCIA*\n\n👤 *${emp}*\n🔘 *MOVIMIENTO:* ${tipo}\n🕒 *HORA:* ${hora}\n📅 *FECHA:* ${fecha}\n📝 *OBS:* ${obs}\n\n✅ _Validado GPS Offline_`; // INTENTAR COPIAR AL PORTAPAPELES (Para que el empleado solo pegue si falla el auto) navigator.clipboard.writeText(texto).then(() => { console.log("Copiado al portapapeles"); }); // LANZAR WHATSAPP enviarWhatsApp(reg); } function enviarWhatsApp(reg) { const texto = `🏢 *LA SECU - ASISTENCIA*\n\n👤 *${reg.emp}*\n🔘 *MOVIMIENTO:* ${reg.tipo}\n🕒 *HORA:* ${reg.hora}\n📅 *FECHA:* ${reg.fecha}\n📝 *OBS:* ${reg.obs}\n\n✅ _Validado GPS Offline_`; // Enlace universal const waUrl = `https://wa.me/${CONFIG.tel}?text=${encodeURIComponent(texto)}`; // Si estamos OFFLINE, avisamos al usuario que ya se copió el reporte if (!navigator.onLine) { alert("⚠️ MODO OFFLINE: El reporte se copió a tu portapapeles. Si WhatsApp no lo pega solo, deja presionado el chat y dale 'PEGAR'."); } window.location.href = waUrl; }
